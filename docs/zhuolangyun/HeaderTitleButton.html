HeaderTitleButton

基于HeaderButton和react-native的View Image Text的封装

规则

  • 统一样式风格的导航器按钮，支持文字和icon

Example

iOS Android

Code

import React, { Component } from 'react';
import {
    HeaderTitleButton,
} from '../../components/theme';



const Navigator = createStackNavigator(
    {
        ListCreate:{
            screen:ListCreate,
            navigationOptions:({ navigation }) => ({
                title:'创建目录',
                headerRight: (
                    <HeaderTitleButton
                        onPress={() => {

                        }}
                        title={'下一步'}
                        icon={require('../assets/list_index_icon_8.png')}
                    />
                ),
            })
        },
    }
)
